076    /**
077     * Client for Distributed Shell application submission to YARN.
078     * 
079     * <p> The distributed shell client allows an application master to be launched that in turn would run 
080     * the provided shell command on a set of containers. </p>
081     * 
082     * <p>This client is meant to act as an example on how to write yarn-based applications. </p>
083     * 
084     * <p> To submit an application, a client first needs to connect to the <code>ResourceManager</code> 
085     * aka ApplicationsManager or ASM via the {@link ApplicationClientProtocol}. The {@link ApplicationClientProtocol} 
086     * provides a way for the client to get access to cluster information and to request for a
087     * new {@link ApplicationId}. <p>
088     * 
089     * <p> For the actual job submission, the client first has to create an {@link ApplicationSubmissionContext}. 
090     * The {@link ApplicationSubmissionContext} defines the application details such as {@link ApplicationId} 
091     * and application name, the priority assigned to the application and the queue
092     * to which this application needs to be assigned. In addition to this, the {@link ApplicationSubmissionContext}
093     * also defines the {@link ContainerLaunchContext} which describes the <code>Container</code> with which 
094     * the {@link ApplicationMaster} is launched. </p>
095     * 
096     * <p> The {@link ContainerLaunchContext} in this scenario defines the resources to be allocated for the 
097     * {@link ApplicationMaster}'s container, the local resources (jars, configuration files) to be made available 
098     * and the environment to be set for the {@link ApplicationMaster} and the commands to be executed to run the 
099     * {@link ApplicationMaster}. <p>
100     * 
101     * <p> Using the {@link ApplicationSubmissionContext}, the client submits the application to the 
102     * <code>ResourceManager</code> and then monitors the application by requesting the <code>ResourceManager</code> 
103     * for an {@link ApplicationReport} at regular time intervals. In case of the application taking too long, the client 
104     * kills the application by submitting a {@link KillApplicationRequest} to the <code>ResourceManager</code>. </p>
105     *
106     */

161      /**
162       * @param args Command line arguments 
163       */
164      public static void main(String[] args) {
165        boolean result = false;
166        try {
167          Client client = new Client();
168          LOG.info("Initializing Client");
169          try {
170            boolean doRun = client.init(args);
171            if (!doRun) {
172              System.exit(0);
173            }
174          } catch (IllegalArgumentException e) {
175            System.err.println(e.getLocalizedMessage());
176            client.printUsage();
177            System.exit(-1);
178          }
179          result = client.run();
180        } catch (Throwable t) {
181          LOG.fatal("Error running CLient", t);
182          System.exit(1);
183        }
184        if (result) {
185          LOG.info("Application completed successfully");
186          System.exit(0);                   
187        } 
188        LOG.error("Application failed to complete successfully");
189        System.exit(2);
190      }

208        opts.addOption("queue", true, "RM Queue in which this application is to be submitted");
209        opts.addOption("timeout", true, "Application timeout in milliseconds");
210        opts.addOption("master_memory", true, "Amount of memory in MB to be requested to run the application master");
211        opts.addOption("jar", true, "Jar file containing the application master");
212        opts.addOption("shell_command", true, "Shell command to be executed by the Application Master");
213        opts.addOption("shell_script", true, "Location of the shell script to be executed");
214        opts.addOption("shell_args", true, "Command line args for the shell script");
215        opts.addOption("shell_env", true, "Environment for shell script. Specified as env_key=env_val pairs");
216        opts.addOption("shell_cmd_priority", true, "Priority for the shell command containers");
217        opts.addOption("container_memory", true, "Amount of memory in MB to be requested to run the shell command");
218        opts.addOption("num_containers", true, "No. of containers on which the shell command needs to be executed");
219        opts.addOption("log_properties", true, "log4j.properties file");

238      /**
239       * Parse command line options
240       * @param args Parsed command line options 
241       * @return Whether the init was successful to run the client
242       * @throws ParseException
243       */
244      public boolean init(String[] args) throws ParseException {
245    
246        CommandLine cliParser = new GnuParser().parse(opts, args);
247    
248        if (args.length == 0) {
249          throw new IllegalArgumentException("No args specified for client to initialize");
250        }           
251    
252        if (cliParser.hasOption("help")) {
253          printUsage();
254          return false;
255        }
256    
257        if (cliParser.hasOption("debug")) {
258          debugFlag = true;
259    
260        }
261    
262        appName = cliParser.getOptionValue("appname", "DistributedShell");
263        amPriority = Integer.parseInt(cliParser.getOptionValue("priority", "0"));
264        amQueue = cliParser.getOptionValue("queue", "default");
265        amMemory = Integer.parseInt(cliParser.getOptionValue("master_memory", "10"));               
266    
267        if (amMemory < 0) {
268          throw new IllegalArgumentException("Invalid memory specified for application master, exiting."
269              + " Specified memory=" + amMemory);
270        }
271    
272        if (!cliParser.hasOption("jar")) {
273          throw new IllegalArgumentException("No jar file specified for application master");
274        }           
275    
276        appMasterJar = cliParser.getOptionValue("jar");
277    
278        if (!cliParser.hasOption("shell_command")) {
279          throw new IllegalArgumentException("No shell command specified to be executed by application master");
280        }
281        shellCommand = cliParser.getOptionValue("shell_command");
282    
283        if (cliParser.hasOption("shell_script")) {
284          shellScriptPath = cliParser.getOptionValue("shell_script");
285        }
286        if (cliParser.hasOption("shell_args")) {
287          shellArgs = cliParser.getOptionValue("shell_args");
288        }
289        if (cliParser.hasOption("shell_env")) { 
290          String envs[] = cliParser.getOptionValues("shell_env");
291          for (String env : envs) {
292            env = env.trim();
293            int index = env.indexOf('=');
294            if (index == -1) {
295              shellEnv.put(env, "");
296              continue;
297            }
298            String key = env.substring(0, index);
299            String val = "";
300            if (index < (env.length()-1)) {
301              val = env.substring(index+1);
302            }
303            shellEnv.put(key, val);
304          }
305        }
306        shellCmdPriority = Integer.parseInt(cliParser.getOptionValue("shell_cmd_priority", "0"));
307    
308        containerMemory = Integer.parseInt(cliParser.getOptionValue("container_memory", "10"));
309        numContainers = Integer.parseInt(cliParser.getOptionValue("num_containers", "1"));
310    
311        if (containerMemory < 0 || numContainers < 1) {
312          throw new IllegalArgumentException("Invalid no. of containers or container memory specified, exiting."
313              + " Specified containerMemory=" + containerMemory
314              + ", numContainer=" + numContainers);
315        }
316    
317        clientTimeout = Integer.parseInt(cliParser.getOptionValue("timeout", "600000"));
318    
319        log4jPropFile = cliParser.getOptionValue("log_properties", "");
320    
321        return true;
322      }

324      /**
325       * Main run function for the client
326       * @return true if application completed successfully
327       * @throws IOException
328       * @throws YarnException
329       */
330      public boolean run() throws IOException, YarnException {
331    
332        LOG.info("Running Client");
333        yarnClient.start();
334    
335        YarnClusterMetrics clusterMetrics = yarnClient.getYarnClusterMetrics();
336        LOG.info("Got Cluster metric info from ASM" 
337            + ", numNodeManagers=" + clusterMetrics.getNumNodeManagers());
338    
339        List<NodeReport> clusterNodeReports = yarnClient.getNodeReports(
340            NodeState.RUNNING);
341        LOG.info("Got Cluster node info from ASM");
342        for (NodeReport node : clusterNodeReports) {
343          LOG.info("Got node report from ASM for"
344              + ", nodeId=" + node.getNodeId() 
345              + ", nodeAddress" + node.getHttpAddress()
346              + ", nodeRackName" + node.getRackName()
347              + ", nodeNumContainers" + node.getNumContainers());
348        }
349    
350        QueueInfo queueInfo = yarnClient.getQueueInfo(this.amQueue);
351        LOG.info("Queue info"
352            + ", queueName=" + queueInfo.getQueueName()
353            + ", queueCurrentCapacity=" + queueInfo.getCurrentCapacity()
354            + ", queueMaxCapacity=" + queueInfo.getMaximumCapacity()
355            + ", queueApplicationCount=" + queueInfo.getApplications().size()
356            + ", queueChildQueueCount=" + queueInfo.getChildQueues().size());               
357    
358        List<QueueUserACLInfo> listAclInfo = yarnClient.getQueueAclsInfo();
359        for (QueueUserACLInfo aclInfo : listAclInfo) {
360          for (QueueACL userAcl : aclInfo.getUserAcls()) {
361            LOG.info("User ACL Info for Queue"
362                + ", queueName=" + aclInfo.getQueueName()                   
363                + ", userAcl=" + userAcl.name());
364          }
365        }           
366    
367        // Get a new application id
368        YarnClientApplication app = yarnClient.createApplication();
369        GetNewApplicationResponse appResponse = app.getNewApplicationResponse();
370        // TODO get min/max resource capabilities from RM and change memory ask if needed
371        // If we do not have min/max, we may not be able to correctly request 
372        // the required resources from the RM for the app master
373        // Memory ask has to be a multiple of min and less than max. 
374        // Dump out information about cluster capability as seen by the resource manager
375        int maxMem = appResponse.getMaximumResourceCapability().getMemory();
376        LOG.info("Max mem capabililty of resources in this cluster " + maxMem);
377    
378        // A resource ask cannot exceed the max. 
379        if (amMemory > maxMem) {
380          LOG.info("AM memory specified above max threshold of cluster. Using max value."
381              + ", specified=" + amMemory
382              + ", max=" + maxMem);
383          amMemory = maxMem;
384        }                           
385    
386        // set the application name
387        ApplicationSubmissionContext appContext = app.getApplicationSubmissionContext();
388        ApplicationId appId = appContext.getApplicationId();
389        appContext.setApplicationName(appName);
390    
391        // Set up the container launch context for the application master
392        ContainerLaunchContext amContainer = Records.newRecord(ContainerLaunchContext.class);
393    
394        // set local resources for the application master
395        // local files or archives as needed
396        // In this scenario, the jar file for the application master is part of the local resources                 
397        Map<String, LocalResource> localResources = new HashMap<String, LocalResource>();
398    
399        LOG.info("Copy App Master jar from local filesystem and add to local environment");
400        // Copy the application master jar to the filesystem 
401        // Create a local resource to point to the destination jar path 
402        FileSystem fs = FileSystem.get(conf);
403        Path src = new Path(appMasterJar);
404        String pathSuffix = appName + "/" + appId.getId() + "/AppMaster.jar";           
405        Path dst = new Path(fs.getHomeDirectory(), pathSuffix);
406        fs.copyFromLocalFile(false, true, src, dst);
407        FileStatus destStatus = fs.getFileStatus(dst);
408        LocalResource amJarRsrc = Records.newRecord(LocalResource.class);
409    
410        // Set the type of resource - file or archive
411        // archives are untarred at destination
412        // we don't need the jar file to be untarred for now
413        amJarRsrc.setType(LocalResourceType.FILE);
414        // Set visibility of the resource 
415        // Setting to most private option
416        amJarRsrc.setVisibility(LocalResourceVisibility.APPLICATION);          
417        // Set the resource to be copied over
418        amJarRsrc.setResource(ConverterUtils.getYarnUrlFromPath(dst)); 
419        // Set timestamp and length of file so that the framework 
420        // can do basic sanity checks for the local resource 
421        // after it has been copied over to ensure it is the same 
422        // resource the client intended to use with the application
423        amJarRsrc.setTimestamp(destStatus.getModificationTime());
424        amJarRsrc.setSize(destStatus.getLen());
425        localResources.put("AppMaster.jar",  amJarRsrc);
426    
427        // Set the log4j properties if needed 
428        if (!log4jPropFile.isEmpty()) {
429          Path log4jSrc = new Path(log4jPropFile);
430          Path log4jDst = new Path(fs.getHomeDirectory(), "log4j.props");
431          fs.copyFromLocalFile(false, true, log4jSrc, log4jDst);
432          FileStatus log4jFileStatus = fs.getFileStatus(log4jDst);
433          LocalResource log4jRsrc = Records.newRecord(LocalResource.class);
434          log4jRsrc.setType(LocalResourceType.FILE);
435          log4jRsrc.setVisibility(LocalResourceVisibility.APPLICATION);        
436          log4jRsrc.setResource(ConverterUtils.getYarnUrlFromURI(log4jDst.toUri()));
437          log4jRsrc.setTimestamp(log4jFileStatus.getModificationTime());
438          log4jRsrc.setSize(log4jFileStatus.getLen());
439          localResources.put("log4j.properties", log4jRsrc);
440        }                   
441    
442        // The shell script has to be made available on the final container(s)
443        // where it will be executed. 
444        // To do this, we need to first copy into the filesystem that is visible 
445        // to the yarn framework. 
446        // We do not need to set this as a local resource for the application 
447        // master as the application master does not need it.               
448        String hdfsShellScriptLocation = ""; 
449        long hdfsShellScriptLen = 0;
450        long hdfsShellScriptTimestamp = 0;
451        if (!shellScriptPath.isEmpty()) {
452          Path shellSrc = new Path(shellScriptPath);
453          String shellPathSuffix = appName + "/" + appId.getId() + "/ExecShellScript.sh";
454          Path shellDst = new Path(fs.getHomeDirectory(), shellPathSuffix);
455          fs.copyFromLocalFile(false, true, shellSrc, shellDst);
456          hdfsShellScriptLocation = shellDst.toUri().toString(); 
457          FileStatus shellFileStatus = fs.getFileStatus(shellDst);
458          hdfsShellScriptLen = shellFileStatus.getLen();
459          hdfsShellScriptTimestamp = shellFileStatus.getModificationTime();
460        }
461    
462        // Set local resource info into app master container launch context
463        amContainer.setLocalResources(localResources);
464    
465        // Set the necessary security tokens as needed
466        //amContainer.setContainerTokens(containerToken);
467    
468        // Set the env variables to be setup in the env where the application master will be run
469        LOG.info("Set the environment for the application master");
470        Map<String, String> env = new HashMap<String, String>();
471    
472        // put location of shell script into env
473        // using the env info, the application master will create the correct local resource for the 
474        // eventual containers that will be launched to execute the shell scripts
475        env.put(DSConstants.DISTRIBUTEDSHELLSCRIPTLOCATION, hdfsShellScriptLocation);
476        env.put(DSConstants.DISTRIBUTEDSHELLSCRIPTTIMESTAMP, Long.toString(hdfsShellScriptTimestamp));
477        env.put(DSConstants.DISTRIBUTEDSHELLSCRIPTLEN, Long.toString(hdfsShellScriptLen));
478    
479        // Add AppMaster.jar location to classpath          
480        // At some point we should not be required to add 
481        // the hadoop specific classpaths to the env. 
482        // It should be provided out of the box. 
483        // For now setting all required classpaths including
484        // the classpath to "." for the application jar
485        StringBuilder classPathEnv = new StringBuilder(Environment.CLASSPATH.$())
486          .append(File.pathSeparatorChar).append("./*");
487        for (String c : conf.getStrings(
488            YarnConfiguration.YARN_APPLICATION_CLASSPATH,
489            YarnConfiguration.DEFAULT_YARN_APPLICATION_CLASSPATH)) {
490          classPathEnv.append(File.pathSeparatorChar);
491          classPathEnv.append(c.trim());
492        }
493        classPathEnv.append(File.pathSeparatorChar).append("./log4j.properties");
494    
495        // add the runtime classpath needed for tests to work
496        if (conf.getBoolean(YarnConfiguration.IS_MINI_YARN_CLUSTER, false)) {
497          classPathEnv.append(':');
498          classPathEnv.append(System.getProperty("java.class.path"));
499        }
500    
501        env.put("CLASSPATH", classPathEnv.toString());
502    
503        amContainer.setEnvironment(env);
504    
505        // Set the necessary command to execute the application master 
506        Vector<CharSequence> vargs = new Vector<CharSequence>(30);
507    
508        // Set java executable command 
509        LOG.info("Setting up app master command");
510        vargs.add(Environment.JAVA_HOME.$() + "/bin/java");
511        // Set Xmx based on am memory size
512        vargs.add("-Xmx" + amMemory + "m");
513        // Set class name 
514        vargs.add(appMasterMainClass);
515        // Set params for Application Master
516        vargs.add("--container_memory " + String.valueOf(containerMemory));
517        vargs.add("--num_containers " + String.valueOf(numContainers));
518        vargs.add("--priority " + String.valueOf(shellCmdPriority));
519        if (!shellCommand.isEmpty()) {
520          vargs.add("--shell_command " + shellCommand + "");
521        }
522        if (!shellArgs.isEmpty()) {
523          vargs.add("--shell_args " + shellArgs + "");
524        }
525        for (Map.Entry<String, String> entry : shellEnv.entrySet()) {
526          vargs.add("--shell_env " + entry.getKey() + "=" + entry.getValue());
527        }                   
528        if (debugFlag) {
529          vargs.add("--debug");
530        }
531    
532        vargs.add("1>" + ApplicationConstants.LOG_DIR_EXPANSION_VAR + "/AppMaster.stdout");
533        vargs.add("2>" + ApplicationConstants.LOG_DIR_EXPANSION_VAR + "/AppMaster.stderr");
534    
535        // Get final commmand
536        StringBuilder command = new StringBuilder();
537        for (CharSequence str : vargs) {
538          command.append(str).append(" ");
539        }
540    
541        LOG.info("Completed setting up app master command " + command.toString());     
542        List<String> commands = new ArrayList<String>();
543        commands.add(command.toString());           
544        amContainer.setCommands(commands);
545    
546        // Set up resource type requirements
547        // For now, only memory is supported so we set memory requirements
548        Resource capability = Records.newRecord(Resource.class);
549        capability.setMemory(amMemory);
550        appContext.setResource(capability);
551    
552        // Service data is a binary blob that can be passed to the application
553        // Not needed in this scenario
554        // amContainer.setServiceData(serviceData);
555    
556        // Setup security tokens
557        if (UserGroupInformation.isSecurityEnabled()) {
558          Credentials credentials = new Credentials();
559          String tokenRenewer = conf.get(YarnConfiguration.RM_PRINCIPAL);
560          if (tokenRenewer == null || tokenRenewer.length() == 0) {
561            throw new IOException(
562              "Can't get Master Kerberos principal for the RM to use as renewer");
563          }
564    
565          // For now, only getting tokens for the default file-system.
566          final Token<?> tokens[] =
567              fs.addDelegationTokens(tokenRenewer, credentials);
568          if (tokens != null) {
569            for (Token<?> token : tokens) {
570              LOG.info("Got dt for " + fs.getUri() + "; " + token);
571            }
572          }
573          DataOutputBuffer dob = new DataOutputBuffer();
574          credentials.writeTokenStorageToStream(dob);
575          ByteBuffer fsTokens = ByteBuffer.wrap(dob.getData(), 0, dob.getLength());
576          amContainer.setTokens(fsTokens);
577        }
578    
579        appContext.setAMContainerSpec(amContainer);
580    
581        // Set the priority for the application master
582        Priority pri = Records.newRecord(Priority.class);
583        // TODO - what is the range for priority? how to decide? 
584        pri.setPriority(amPriority);
585        appContext.setPriority(pri);

587        // Set the queue to which this application is to be submitted in the RM
588        appContext.setQueue(amQueue);
589    
590        // Submit the application to the applications manager
591        // SubmitApplicationResponse submitResp = applicationsManager.submitApplication(appRequest);
592        // Ignore the response as either a valid response object is returned on success 
593        // or an exception thrown to denote some form of a failure
594        LOG.info("Submitting application to ASM");

596        yarnClient.submitApplication(appContext);
597    
598        // TODO
599        // Try submitting the same request again

607      /**
608       * Monitor the submitted application for completion. 
609       * Kill application if time expires. 
610       * @param appId Application Id of application to be monitored
611       * @return true if application completed successfully
612       * @throws YarnException
613       * @throws IOException
614       */
615      private boolean monitorApplication(ApplicationId appId)
616          throws YarnException, IOException {
617    
618        while (true) {
619    
620          // Check app status every 1 second.
621          try {
622            Thread.sleep(1000);
623          } catch (InterruptedException e) {
624            LOG.debug("Thread sleep in monitoring loop interrupted");
625          }
626    
627          // Get application report for the appId we are interested in 
628          ApplicationReport report = yarnClient.getApplicationReport(appId);
629    
630          LOG.info("Got application report from ASM for"
631              + ", appId=" + appId.getId()
632              + ", clientToAMToken=" + report.getClientToAMToken()
633              + ", appDiagnostics=" + report.getDiagnostics()
634              + ", appMasterHost=" + report.getHost()
635              + ", appQueue=" + report.getQueue()
636              + ", appMasterRpcPort=" + report.getRpcPort()
637              + ", appStartTime=" + report.getStartTime()
638              + ", yarnAppState=" + report.getYarnApplicationState().toString()
639              + ", distributedFinalState=" + report.getFinalApplicationStatus().toString()
640              + ", appTrackingUrl=" + report.getTrackingUrl()
641              + ", appUser=" + report.getUser());
642    
643          YarnApplicationState state = report.getYarnApplicationState();
644          FinalApplicationStatus dsStatus = report.getFinalApplicationStatus();
645          if (YarnApplicationState.FINISHED == state) {
646            if (FinalApplicationStatus.SUCCEEDED == dsStatus) {
647              LOG.info("Application has completed successfully. Breaking monitoring loop");
648              return true;        
649            }
650            else {
651              LOG.info("Application did finished unsuccessfully."
652                  + " YarnState=" + state.toString() + ", DSFinalStatus=" + dsStatus.toString()
653                  + ". Breaking monitoring loop");
654              return false;
655            }                         
656          }
657          else if (YarnApplicationState.KILLED == state     
658              || YarnApplicationState.FAILED == state) {
659            LOG.info("Application did not finish."
660                + " YarnState=" + state.toString() + ", DSFinalStatus=" + dsStatus.toString()
661                + ". Breaking monitoring loop");
662            return false;
663          }                 
664    
665          if (System.currentTimeMillis() > (clientStartTime + clientTimeout)) {
666            LOG.info("Reached client specified timeout for application. Killing application");
667            forceKillApplication(appId);
668            return false;                           
669          }
670        }                   
671    
672      }
